script
   Highlight selected category tile
  document.addEventListener(click, function(e) {
    if (e.target.closest(.category-tile)) {
      document.querySelectorAll(.category-tile).forEach(t = t.classList.remove(selected));
      e.target.closest(.category-tile).classList.add(selected);
      const radio = e.target.closest(.category-tile).querySelector(input[type='radio']);
      if (radio) radio.checked = true;
    }
  });
